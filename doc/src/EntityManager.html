246    def get_current_animation(self):
247        """Returns the current animation."""
248        # Check if entity is visible
249        if (
250            self._is_visible is False
251            or self.animations[self.current_animation_type] == []
252        ):
253            return None
254
255        current_asset = self.animations[self.current_animation_type][
256            int(self.current_animation_index)
257        ]
258
259        # Check if there is an image to display
260        if current_asset is None:
261            return None
262
263        # Update current frame
264        animation_speed = (
265            self.animation_speed[self.current_animation_type]
266            if hasattr(self, "animation_speed")
267            and self.current_animation_type in self.animation_speed
268            else 1
269        )

271            self.current_animation_index + animation_speed
272        ) % len(self.animations[self.current_animation_type])
273
274        # Rotate frame
275        current_asset = current_asset.rotate(self.rotation)
276
277        # Resize frame
278        current_asset = current_asset.scale(self.size)
279
280        # Reverse frame
281        current_asset = current_asset.reverse(self.reverse)
282
283        return current_asset
